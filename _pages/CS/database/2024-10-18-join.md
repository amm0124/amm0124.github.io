두 쿼리의 차이에 대해 알아보자

    create table student_ppt07(
        name varchar(255),
        gpa numeric(4,0)
    );

    INSERT INTO student_ppt07 (name, gpa) VALUES ('Mary', 3.8);
    INSERT INTO student_ppt07 (name, gpa) VALUES ('Tom', 3.6);
    INSERT INTO student_ppt07 (name, gpa) VALUES ('Jack', 3.7);

    create table enroll_ppt07(
        stdName varchar(255),
        course numeric(4,0)
    );

    insert into enroll_ppt07(stdName, course) VALUES('Mary', 354);
    insert into enroll_ppt07(stdName, course) VALUES('Tom', 354);
    insert into enroll_ppt07(stdName, course) VALUES('Tom', 454);
    insert into enroll_ppt07(stdName, course) VALUES('Bob', 354);

    select name, course
    from student_ppt07 left join enroll_ppt07 on
    name = stdname and course=354;

    select name, course
    from student_ppt07 left join enroll_ppt07 on
    name = stdname 
    where course=354
