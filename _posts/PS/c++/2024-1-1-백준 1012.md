# 백준 1012 (유기농 배추) 

## 문제 링크 

[문제 링크 : https://www.acmicpc.net/problem/1012](https://www.acmicpc.net/problem/1012)

## 문제 접근

간단한 탐색 문제입니다. 
2차원 배열의 형태로 밭이 주어집니다. 배추가 있는 곳은 1로, 배추가 없는 그냥 땅은 0으로 입력이 주어집니다.
배추흰지렁이가 배추가 심어진 땅을 12,3,6,9시 방향으로 확산이 가능하다고 합니다.
배추흰지렁이가 확보하지 않은 배추 위치를 기준으로, 배추흰지렁이를 둡니다. 이후 4방향 탐색(dx, dy)을 통하여 배추흰지렁이가 해충을 지킬 수 있는 영역을 확보하고, 확보한 영역을 저는 0도 1도 아닌 2로 설정을 하였습니다. <br>
bool 형태의 nXm 배열을 선언한 후, 방문했으면 true, 방문하지 않았으면 false로 설정해도 됩니다. <br>
이후 배추흰지렁이의 수를 세면 됩니다.

![image](https://github.com/amm0124/amm0124.github.io/assets/108533909/3eb0bcdf-3552-4b26-b200-5e497d864764)

저는 stack을 사용하는 dfs로 구현을 했습니다. 평소에는 재귀함수로 짜는 것을 좋아하는데, 다른 문제를 풀 때마다, 재귀로 짜니 시간 초과가 나는 경우가 많더라고요. 특히 dp문제에서 시간 초과가 나는 경우가 있었습니다. 그래서 이번에는 stack을 사용해서 dfs를 구현해보았습니다.

그리고 입출력 성능 향상을 위해서 버퍼 동기화 해제 및 버퍼 비우기 해제를 하니 속도가 4ms에서 0ms로 개선이 된 모습을 확인할 수 있었습니다.

## 소스 코드

    #include <iostream>
    #include <vector>
    #include <stack>

    using namespace std;

    int main(void){
        ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL); //입출력 향상
        int testCase;

        cin>>testCase;

        for(int i=0;i<testCase;i++){
            int m,n,k;
            cin>>m>>n>>k;
            int answer=0;
            vector<vector<int>> field;
            vector<int> v(m,0);
            for(int j=0;j<n;j++){
                field.push_back(v);
            } //밭 만들기. 초기 상태는 0으로 초기화

            vector<pair<int,int>> positions; 
            for(int j=0;j<k;j++){
                int pos_x,pos_y;
                cin>>pos_x>>pos_y;
                field[pos_y][pos_x]=1; // 밭에 배추 심기. 
                positions.push_back(make_pair(pos_y,pos_x)); //위치 기억해서 바로 시작
            }

            for(auto position : positions){ //배추 심은 위치만 기억해서 반복하기.
                stack<pair<int,int>> s; //dfs용 stack

                if(field[position.first][position.second]!=1){ //2인 경우
                    continue;
                }else{ //1인 경우
                    answer++;
                    s.push(position);
                    int dx[4]={-1,0,1,0};
                    int dy[4]={0,1,0,-1}; //4방향 dfs.
                    while(!s.empty()){
                        int pos_x = s.top().first;
                        int pos_y = s.top().second;
                        s.pop();
                        field[pos_x][pos_y]=2;
                        // 다음 땅이 배추가 심어져 있지 않거나, 배추를 심고 방문한 땅이라면
                        for(int j=0;j<4;j++){
                            int next_pos_x = pos_x + dx[j];
                            int next_pos_y = pos_y + dy[j];
                            //범위 내에 있으면서 
                            if(next_pos_x>=0 && next_pos_x<=n-1 && next_pos_y>=0 && next_pos_y<=m-1){
                                if(field[next_pos_x][next_pos_y]==1){
                                    s.push(make_pair(next_pos_x,next_pos_y));
                                }
                            }
                        }
                    }
                }

            }
            cout<<answer<<endl; //정답 출력
        }
        return 0;
    }
